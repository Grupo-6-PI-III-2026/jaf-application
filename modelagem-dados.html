Sistema de Gestão de Obras - JAF Application

📊 Visão Geral do Sistema

O JAF Application é um sistema completo para gestão de obras da construtora JAF, permitindo o controle de funcionários, alocações, gastos, presença e relatórios. O sistema utiliza uma arquitetura baseada em Spring Boot no backend e React no frontend, com persistência de dados em MySQL.

💾 Estrutura do Banco de Dados

🗃️ Schema SQL - MySQL

-- Tabela de Funcionários
CREATE TABLE funcionario (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255),
    email VARCHAR(255),
    senha VARCHAR(255),
    foto_url VARCHAR(512),
    cargo_global VARCHAR(50)
);

-- Tabela de Obras
CREATE TABLE obra (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255),
    orcamento VARCHAR(255),
    status VARCHAR(255),
    dt_inicio DATE,
    dt_termino_previsto DATE
);

-- Tabela de Alocação de Funcionários em Obras
CREATE TABLE alocacao_obra (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    cargo VARCHAR(50),
    funcionario_id BIGINT,
    obra_id BIGINT,
    FOREIGN KEY (funcionario_id) REFERENCES funcionario(id),
    FOREIGN KEY (obra_id) REFERENCES obra(id)
);

-- Tabela de Gastos
CREATE TABLE gasto (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(255),
    categoria VARCHAR(255),
    metodo_pagamento VARCHAR(255),
    etapa VARCHAR(255),
    valor DECIMAL(19, 2),
    dt_gasto DATE,
    reembolso_concluido BOOLEAN,
    funcionario_id BIGINT,
    obra_id BIGINT,
    FOREIGN KEY (funcionario_id) REFERENCES funcionario(id),
    FOREIGN KEY (obra_id) REFERENCES obra(id)
);

-- Tabela de Presença
CREATE TABLE presenca (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    funcionario_id BIGINT,
    obra_id BIGINT,
    data DATE,
    presente BOOLEAN,
    horario_entrada TIME,
    horario_saida TIME,
    FOREIGN KEY (funcionario_id) REFERENCES funcionario(id),
    FOREIGN KEY (obra_id) REFERENCES obra(id)
);

-- Tabela de Relatórios
CREATE TABLE relatorio (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255),
    dt_emissao DATE,
    funcionario_id BIGINT,
    FOREIGN KEY (funcionario_id) REFERENCES funcionario(id)
);
